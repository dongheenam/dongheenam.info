{{/* Usage: {{% img src="fig-link" alt="alt-text" %}} */}} {{/* if src is an external link, attach it as is */}} {{- if in (.Get "src") "http" -}} {{- else -}} {{/* src should point to a resource */}} {{- $img := resources.Get (.Get "src") -}} {{/* if vector graphics then attach it as is */}} {{- if in $img.RelPermalink "svg" -}} {{- else -}} {{- $img_options := dict "image" $img "alt" (.Get "alt") -}} {{- partial "resp-image" $img_options -}} {{- end -}} {{- end -}}
