{{/* Automatically resizes images and generates an srcset of images */}} {{/* Usage: {{% resp-image image="path" alt="alt text" style="optional css style" class="optional css class" location="main|fullwidth|sidenote" %}} */}} {{/* generating images */}} {{- $default_image := "" -}}{{- $original := "" -}} {{- $tiny := "" -}}{{- $small := "" -}}{{- $medium := "" -}}{{- $large := "" -}} {{- if ge .image.Width "480" -}} {{- $tiny = .image.Resize "480x q100 webp" -}} {{- $default_image = .image.Resize "480x q100" -}} {{- else -}} {{- $default_image = .image -}} {{- end -}} {{- if ge .image.Width "768" -}} {{- $small = .image.Resize "768x q100 webp" -}} {{- end -}} {{- if ge .image.Width "1024" -}} {{- $medium = .image.Resize "1024x q100 webp" -}} {{- end -}} {{- if ge .image.Width "2048" -}} {{- $large = .image.Resize "2048x q100 webp" -}} {{- end -}} {{ $original = .image.Resize (printf "%dx q100 webp" .image.Width) }} {{/* defining sizes */}} {{- $sizes := "" -}} {{- if eq .loc "sidenote" -}} {{ $sizes = "(min-width: 1620px) 375px, (min-width: 1220px) 225px, (min-width: 780px) 85vw, 95vw" }} {{- else -}} {{ $sizes = "100vw" }} {{- end -}} {{/* printing sources */}} {{- $srcset := "" -}} {{- if ne $tiny "" }}{{ $srcset = printf "%s 480w" $tiny.RelPermalink }}{{ end }} {{- if ne $small "" }}{{ $srcset = printf "%s, %s 768w" $srcset $small.RelPermalink }}{{ end }} {{- if ne $medium "" }}{{ $srcset = printf "%s, %s 1024w" $srcset $medium.RelPermalink }}{{ end }} {{- if ne $large "" }}{{ $srcset = printf "%s, %s 2048w" $srcset $large.RelPermalink }}{{ end }} {{ $srcset = printf "%s, %s %dw" $srcset $original.RelPermalink .image.Width }}
