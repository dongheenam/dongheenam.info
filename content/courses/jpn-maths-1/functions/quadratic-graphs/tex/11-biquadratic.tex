\newcommand{\xmax}{  7 }
\newcommand{\ymin}{ -2 }
\newcommand{\ymax}{ 10 }

\begin{document}
\begin{tikzpicture}

\begin{axis}[without_grid,
    set layers=axis on top,
    xmin=\xmin,
    xmax=\xmax,
    ymin=\ymin,
    ymax=\ymax,
    height=6.5cm]

% origin
\node [anchor=north east] at (0,0) {$O$};

% top and bottom lines (for filling)
\path[name path=top] (\xmin, \ymax) -- (\xmax, \ymax);
\path[name path=bot] (\xmin, \ymin) -- (\xmax, \ymin);
\addplot[blue!20] fill between[of=top and bot, soft clip={domain=0:7}];

% graph
\addplot [domain=-1:7, samples=100, thick, dotted] {0.8*(x^2-6*x+10)};
\addplot [domain=0:7, samples=100, thick] {0.8*(x^2-6*x+10)};

% coordinates
\addplot [dashed] coordinates {(0,0.8)(3,0.8)} node[pos=0, anchor=east] {$1$};
\addplot [mark=*] coordinates {(0,8)} node[pos=0, anchor=east] {$10$};
\addplot [dashed] coordinates {(3,0)(3,0.8)} node[pos=0, anchor=north] {$3$};

\addplot [red, mark=*] coordinates {(3,0.8)} node[anchor=160] {min};

\node [circle,inner sep=0pt,minimum size=12pt,draw,fill=white,semithick] at (-1.3, 9.3) {$1$};

\end{axis}
\end{tikzpicture}

\end{document}
